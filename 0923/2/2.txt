16 bit
MSB = 1

#include <stdio.h>

int main() {
    unsigned short a = 0b0000000000000000;  // 2바이트 (16비트)로 초기화

    // MSB를 1로 세팅
    a = a | (0b1 << 15); // 15번째 비트에 1을 설정

    // 2번 비트부터 5번 비트까지 1로 세팅
    a = a | (0b111 << 2); // 2~5 비트에 1을 설정

    // 8번 ~ 13번 비트까지 1로 세팅
    a = a | (0b111111 << 8); // 8~13 비트에 1을 설정

    // 3번 ~ 7번 비트를 반전(~)시키기
    a = a ^ (0b1111110 << 2); // 2비트부터 7비트까지 반전

    // 최상위 3비트를 모두 0으로 세팅
    a = a & 0b0111111111111111; // 최상위 3비트 0으로 설정

    printf("결과: %x\n", a); // 결과 출력

    return 0;
}

결과: 3ec4
